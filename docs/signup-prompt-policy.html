// 결정론적 해시 기반 — userId 없으면 deviceId(UUID) 생성 후 localStorage 저장
function assignVariant(seed: string): 'A' | 'B' | 'C' {
  const hash = simpleHash(seed) // 32bit unsigned int
  const bucket = hash % 3
  return bucket === 0 ? 'A' : bucket === 1 ? 'B' : 'C'
}

6샘플 사이즈 & 통계 설계
v3 수정: 샘플 기준을 "20초 AND 스크롤 50%"로 통일. v2의 Section 2(20초만)와 Section 6(20초+스크롤) 불일치를 해소.
기준 가정
  • 베이스라인 CVR: 3% (현재 비회원 → 가입 전환 추정)
  • 최소 감지 효과: +1.5%p (3% → 4.5%, 상대 50% 개선)
  • 통계적 유의수준: α = 0.05 (단측)
  • 검정력: 80%
  • 배리언트 수: 3 (A/B/C)
필요 샘플 수
배리언트당전체 (3개)예상 수집 기간
약 1,200명약 3,600명DAU 300명 기준 약 12일
집계 단위

"20초 AND 스크롤 50%" 조건을 충족한 비회원 세션 기준. 조건 미충족 세션은 분모에서 제외.

집계 기준: eligible vs shown
지표분모설명
노출률eligible조건 충족자 중 실제로 배너를 본 비율 (차단 영향 측정)
CTRshown배너를 본 사람 중 클릭률
CVR (핵심)eligible조건 충족자 중 가입 전환율 — A/B 비교의 진짜 기준
승자 결정 기준
  • Fisher's exact test p < 0.05
  • 3개 배리언트 동시 비교 시 Bonferroni 보정 적용 (α/3 = 0.017)
  • CVR 차이가 +1%p 미만이면 실용적 유의성 없음으로 간주 → 롤아웃 보류
⚠️ 갭 C — 교차 디바이스 오염 한계
unao_device_id(localStorage)로 배리언트 고정 → 같은 사람이 모바일+PC로 접속하면 다른 배리언트 배정 가능.

수용 근거: 우나어 트래픽 규모(DAU ~300명)에서는 교차 디바이스 비율이 낮아 통계적 오염 영향 미미. 로그인 후 user_id 기반 고정은 v2 테스트 이후 적용 검토.
분석 시 주의: GA4 보고서에서 user_id 기준과 device_id 기준 결과를 교차 확인할 것.

8PWA 유도 정책 연계
pwa_page_views_after_signup 카운터
⚠️ GAP 2 해소
v2에는 이 카운터 구현 스펙이 없었음. 어떤 컴포넌트도 카운트하지 않아 "가입 후 3페이지 탐색 시 PWA 유도" 조건이 실제로 작동하지 않았음.
✅ 해결 방안
layout.tsx의 useEffect에서 카운팅. pathname 변경 시마다 CONTENT_PATHS 해당 여부 확인 후 increment.
// src/app/(main)/layout.tsx 내 useEffect
const CONTENT_PATHS = ['/community/', '/magazine/', '/jobs/']

useEffect(() => {
  if (!session?.user?.id) return // 비회원은 카운트 안 함
  const completedAt = localStorage.getItem('signup_completed_at')
  if (!completedAt) return // 가입 완료 플래그 없으면 스킵
  if (!CONTENT_PATHS.some(p => pathname.startsWith(p))) return
  const current = parseInt(localStorage.getItem('pwa_page_views_after_signup') || '0')
  localStorage.setItem('pwa_page_views_after_signup', String(current + 1))
}, [pathname, session])
카운터 리셋 조건
  • PWA 설치 완료 시 (pwa_installed='1' 저장 시) → 카운터 삭제
  • signup_completed_at이 없으면 카운팅 자체를 시작하지 않음

9배너-PWA 타이밍 충돌 방지

10비회원 판단 & 차단 우선순위
판단 우선순위 (AND 조건, 위에서 아래로)

  • 4순위 — 영구 차단: signup_prompt_done='1' 있음 → best-effort 영구 차단
  • 5순위 — 트리거 조건: 20초 AND 스크롤 50% 미충족 → 배너 미노출

Tab Visibility API 타이머 구현
// 탭 비활성 시 타이머 일시정지 구현 패턴
let elapsed = 0 // 누적 활성 시간(ms)
let startTime: number // 현재 활성 시작 시각
let rafId: number // requestAnimationFrame ID

function tick() {
  if (document.hidden) return // 비활성 시 건너뜀
  elapsed = Date.now() - startTime
  if (elapsed >= BANNER_DELAY_MS && scrollReached) {
    showBanner()
    return // RAF 정지
  }
  rafId = requestAnimationFrame(tick)
}

document.addEventListener('visibilitychange', () => {
  if (document.hidden) {
    cancelAnimationFrame(rafId) // 일시정지
  } else {
    startTime = Date.now() - elapsed // 경과 시간 보정
    rafId = requestAnimationFrame(tick) // 재개
  }
})

// useEffect cleanup: pathname 변경 시 타이머 + 스크롤 상태 초기화
return () => {
  cancelAnimationFrame(rafId)
  elapsed = 0
  scrollReached = false
}
